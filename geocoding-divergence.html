# Helper: build line sfc from a divergent subset
make_lines <- function(df) {
  lapply(seq_len(nrow(df)), function(i) {
    row <- df[i, ]
    st_linestring(matrix(
      c(row$google_lon, row$source_lon, row$google_lat, row$source_lat),
      ncol = 2
    ))
  }) |> st_sfc(crs = 4326)
}

# Helper: add a recommendation layer group (line + preferred point + grey rejected point)
add_rec_layer <- function(map, df, group, preferred_color) {
  if (nrow(df) == 0) return(map)
  prefer_google <- group == "Rec: Google"

  map |>
    # Connecting line
    addPolylines(
      data = st_sf(df, geometry = make_lines(df)),
      color = preferred_color, weight = 2, opacity = 0.5,
      group = group, popup = ~popup_html
    ) |>
    # Preferred point (colored)
    addCircleMarkers(
      data = df,
      lng = if (prefer_google) ~google_lon else ~source_lon,
      lat = if (prefer_google) ~google_lat else ~source_lat,
      radius = 7, color = preferred_color, fillColor = preferred_color,
      fillOpacity = 0.9, weight = 2, opacity = 0.9,
      group = group, label = ~search_label, popup = ~popup_html
    ) |>
    # Rejected point (grey)
    addCircleMarkers(
      data = df,
      lng = if (prefer_google) ~source_lon else ~google_lon,
      lat = if (prefer_google) ~source_lat else ~google_lat,
      radius = 5, color = "#999", fillColor = "#bbb",
      fillOpacity = 0.6, weight = 1, opacity = 0.6,
      group = group, label = ~search_label, popup = ~popup_html
    )
}

# Classify divergent facilities by data presence
facility_presence <- tar_read(facility_presence)
source_presence <- tar_read(source_presence)
facilities_panel <- tar_read(facilities_panel)
ddp_fy25_summary <- tar_read(ddp_fy25_summary)
detloc_lookup <- tar_read(detloc_lookup)

# FY26 presence from facility_presence (panel IDs 1-398)
fy26_ids <- facility_presence |>
  filter(FY26) |>
  pull(canonical_id)

# DDP presence from source_presence
ddp_ids <- source_presence |>
  filter(in_ddp) |>
  pull(canonical_id)

# Peak ADP from panel data (max across all fiscal years)
peak_adp_panel <- facilities_panel |>
  group_by(canonical_id) |>
  summarise(peak_adp_panel = max(adp, na.rm = TRUE), .groups = "drop") |>
  mutate(peak_adp_panel = ifelse(is.infinite(peak_adp_panel), NA_real_, peak_adp_panel))

# DDP ADP via detloc_lookup join
ddp_adp <- ddp_fy25_summary |>
  inner_join(detloc_lookup, by = c("detention_facility_code" = "detloc")) |>
  group_by(canonical_id) |>
  summarise(
    ddp_adp = max(adp_total, na.rm = TRUE),
    ddp_peak = max(peak_population, na.rm = TRUE),
    .groups = "drop"
  )

divergent_classified <- divergent |>
  mutate(
    in_fy26 = canonical_id %in% fy26_ids,
    in_ddp = canonical_id %in% ddp_ids,
    presence = case_when(
      in_fy26 & in_ddp  ~ "Both FY26 + DDP",
      in_fy26 & !in_ddp ~ "FY26 only",
      !in_fy26 & in_ddp ~ "DDP only",
      TRUE               ~ "Neither"
    )
  ) |>
  left_join(peak_adp_panel, by = "canonical_id") |>
  left_join(ddp_adp, by = "canonical_id") |>
  mutate(
    # Best available ADP: panel peak, then DDP peak
    best_peak_adp = coalesce(peak_adp_panel, ddp_peak)
  )
